CREATE tree-right MAX_NODES ALLOT
CREATE tree-axis MAX_NODES ALLOT

VARIABLE node-count
VARIABLE root-node

: tree-init ( -- )
    0 node-count !
    0 root-node ! ;

\ Insert a point into the tree
: tree-insert ( x y z -- )
    node-count @ root-node @ 0= IF
        \ First node - make it root
        tree-data 3 * 0 3 * MOVE
        0 tree-left 0 1 * MOVE
        0 tree-right 0 1 * MOVE
        0 tree-axis 0 1 * MOVE

        0 root-node !

        \ Find insertion point
        tree-data 3 * 0 3 * MOVE
        0 tree-left 0 1 * MOVE
