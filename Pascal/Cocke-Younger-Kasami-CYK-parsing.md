# Cocke-Younger-Kasami (CYK) Parsing Algorithm in Pascal

Here's a complete implementation of the CYK parsing algorithm in Pascal:

```pascal
program CYKParser;

uses
  SysUtils, Classes;

type
  // Grammar rule structure
  TRule = record
    Left: string;
    Right1, Right2: string;
  end;

  // Grammar structure
  TGrammar = record
    NonTerminals: array[1..10] of string;
    Terminals: array[1..10] of string;
    Rules: array[1..20] of TRule;
    NumRules: integer;
    NumNonTerminals: integer;
    NumTerminals: integer;
  end;

  // CYK table structure
  TCYKTable = array[1..20, 1..20] of array[1..10] of string;

var
  Grammar: TGrammar;
  CYKTable: TCYKTable;
  InputString: string;
  StringLength: integer;

// Function to check if a symbol is a terminal
function IsTerminal(S: string): boolean;
var
  i: integer;
begin
  for i := 1 to Grammar.NumTerminals do
    if Grammar.Terminals[i] = S then
    begin
      IsTerminal := true;
      exit;
    end;
  IsTerminal := false;
end;

// Function to check if a symbol is a non-terminal
function IsNonTerminal(S: string): boolean;
var
  i: integer;
begin
  for i := 1 to Grammar.NumNonTerminals do
    if Grammar.NonTerminals[i] = S then
    begin
      IsNonTerminal := true;
      exit;
    end;
  IsNonTerminal := false;
end;

// Function to check if a non-terminal generates a terminal
function GeneratesTerminal(NonTerm, Terminal: string): boolean;
var
  i: integer;
begin
  for i := 1 to Grammar.NumRules do
    if (Grammar.Rules[i].Left = NonTerm) and 
       (Grammar.Rules[i].Right1 = Terminal) and
       (Grammar.Rules[i].Right2 = '') then
    begin
      GeneratesTerminal := true;
      exit;
    end;
  GeneratesTerminal := false;
end;

// Function to check if a non-terminal generates two non-terminals
function GeneratesNonTerminals(NonTerm, NT1, NT2: string): boolean;
var
  i: integer;
begin
  for i := 1 to Grammar.NumRules do
    if (Grammar.Rules[i].Left = NonTerm) and 
       (Grammar.Rules[i].Right1 = NT1) and
       (Grammar.Rules[i].Right2 = NT2) then
    begin
      GeneratesNonTerminals := true;
      exit;
    end;
  GeneratesNonTerminals := false;
end;

// Function to check if a symbol is in the CYK table cell
function IsInCell(Cell: array of string; Symbol: string): boolean;
var
  i: integer;
begin
  for i := 1 to Length(Cell) do
    if Cell[i] = Symbol then
    begin
      IsInCell := true;
      exit;
    end;
  IsInCell := false;
end;

// Function to add symbol to CYK table cell
procedure AddToCell(var Cell: array of string; var CellCount: integer; Symbol: string);
var
  i: integer;
begin
  for i := 1 to CellCount do
    if Cell[i] = Symbol then
      exit;
  
  if CellCount < 10 then
  begin
    CellCount := CellCount + 1;
    Cell[CellCount] := Symbol;
  end;
end;

// CYK Parsing Algorithm
procedure CYKParse(S: string);
var
  i, j, k, l, cellCount: integer;
  symbol1, symbol2: string;
  tempCell: array[1..10] of string;
  tempCount: integer;
begin
  StringLength := Length(S);
  
  // Initialize CYK table
  for i := 1 to StringLength do
    for j := 1 to StringLength do
      for k := 1 to 10 do
        CYKTable[i, j][k] := '';
  
  // Step 1: Fill the diagonal (length 1)
  for i := 1 to StringLength do
  begin
    // Check which non-terminals generate the terminal at position i
    for j := 1 to Grammar.NumRules do
    begin
      if (Grammar.Rules[j].Right1 = S[i]) and (Grammar.Rules[j].Right2 = '') then
      begin
        // Add the left non-terminal to the table
        cellCount := 0;
        for k := 1 to 10 do
          if CYKTable[i, i][k] <> '' then
            cellCount := cellCount + 1;
        AddToCell(CYKTable[i, i], cellCount, Grammar.Rules[j].Left);
      end;
    end;
  end;
  
  // Step 2: Fill the table for substrings of increasing length
  for l := 2 to StringLength do  // l is the length of the substring
  begin
    for i := 1 to (StringLength - l + 1) do  // i is the starting position
    begin
      j := i + l - 1;  // j is the ending position
      
      // For each possible split point k
      for k := i to (j - 1) do
      begin
        // Check each rule to see if it can be applied
        for var ruleIndex := 1 to Grammar.NumRules do
        begin
          if (Grammar.Rules[ruleIndex].Right2 <> '') and  // It's a binary rule
             (IsNonTerminal(Grammar.Rules[ruleIndex].Right1)) and
             (IsNonTerminal(Grammar.Rules[ruleIndex].Right2)) then
          begin
            // Check if the left part matches the left part of the rule
            symbol1 := Grammar.Rules[ruleIndex].Right1;
            symbol2 := Grammar.Rules[ruleIndex].Right2;
            
            // Check if the left part of the rule can be generated by the left substring
            if (CYKTable[i, k][1] <> '') and (CYKTable[k+1, j][1] <> '') then
            begin
              // Find if symbol1 is in CYKTable[i, k]
              for var m := 1 to 10 do
                if CYKTable[i, k][m] <> '' then
                begin
                  if CYKTable[i, k][m] = symbol1 then
                  begin
                    // Find if symbol2 is in CYKTable[k+1, j]
                    for var n := 1 to 10 do
                      if CYKTable[k+1, j][n] <> '' then
                      begin
                        if CYKTable[k+1, j][n] = symbol2 then
                        begin
                          // Add the left non-terminal to the current cell
                          cellCount := 0;
                          for var p := 1 to 10 do
                            if CYKTable[i, j][p] <> '' then
                              cellCount := cellCount + 1;
                          AddToCell(CYKTable[i, j], cellCount, Grammar.Rules[ruleIndex].Left);
                        end;
                      end;
                  end;
                end;
            end;
          end;
        end;
      end;
    end;
  end;
end;

// Function to print the CYK table
procedure PrintCYKTable;
var
  i, j: integer;
begin
  writeln('CYK Parsing Table:');
  writeln('==================');
  
  for i := 1 to StringLength do
  begin
    for j := 1 to StringLength do
    begin
      write('T[');
      write(i, ',', j, '] = ');
      if CYKTable[i, j][1] <> '' then
      begin
        write('{');
        var k := 1;
        while (k <= 10) and (CYKTable[i, j][k] <> '') do
        begin
          write(CYKTable[i, j][k]);
          if (k < 10) and (CYKTable[i, j][k+1] <> '') then
            write(', ');
          inc(k);
        end;
        writeln('}');
      end
      else
        writeln('Empty');
    end;
    writeln;
  end;
end;

// Function to check if input string is accepted
function IsAccepted: boolean;
begin
  if StringLength > 0 then
  begin
    // Check if start symbol is in the top-right cell
    for var i := 1 to 10 do
      if CYKTable[1, StringLength][i] = 'S' then
      begin
        IsAccepted := true;
        exit;
      end;
  end;
  IsAccepted := false;
end;

// Main procedure
procedure InitializeGrammar;
begin
  // Define grammar (Example: S -> AB | BC, A -> BA | a, B -> CC | b, C -> AB | a)
  Grammar.NumNonTerminals := 3;
  Grammar.NonTerminals[1] := 'S';
  Grammar.NonTerminals[2] := 'A';
  Grammar.NonTerminals[3] := 'B';
  
  Grammar.NumTerminals := 2;
  Grammar.Terminals[1] := 'a';
  Grammar.Terminals[2] := 'b';
  
  Grammar.NumRules := 7;
  
  // Rule 1: S -> AB
  Grammar.Rules[1].Left := 'S';
  Grammar.Rules[1].Right1 := 'A';
  Grammar.Rules[1].Right2 := 'B';
  
  // Rule 2: S -> BC
  Grammar.Rules[2].Left := 'S';
  Grammar.Rules[2].Right1 := 'B';
  Grammar.Rules[2].Right2 := 'C';
  
  // Rule 3: A -> BA
  Grammar.Rules[3].Left := 'A';
  Grammar.Rules[3].Right1 := 'B';
  Grammar.Rules[3].Right2 := 'A';
  
  // Rule 4: A -> a
  Grammar.Rules[4].Left := 'A';
  Grammar.Rules[4].Right1 := 'a';
  Grammar.Rules[4].Right2 := '';
  
  // Rule 5: B -> CC
  Grammar.Rules[5].Left := 'B';
  Grammar.Rules[5].Right1 := 'C';
  Grammar.Rules[5].Right2 := 'C';
  
  // Rule 6: B -> b
  Grammar.Rules[6].Left := 'B';
  Grammar.Rules[6].Right1 := 'b';
  Grammar.Rules[6].Right2 := '';
  
  // Rule 7: C -> AB
  Grammar.Rules[7].Left := 'C';
  Grammar.Rules[7].Right1 := 'A';
  Grammar.Rules[7].Right2 := 'B';
end;

begin
  writeln('Cocke-Younger-Kasami (CYK) Parsing Algorithm');
  writeln('============================================');
  
  InitializeGrammar;
  
  // Test with input string
  InputString := 'abab';
  writeln('Input string: ', InputString);
  writeln;
  
  // Perform CYK parsing
  CYKParse(InputString);
  
  // Print the CYK table
  PrintCYKTable;
  
  // Check if the string is accepted
  if IsAccepted then
    writeln('Result: String "', InputString, '" is ACCEPTED by the grammar')
  else
    writeln('Result: String "', InputString, '" is REJECTED by the grammar');
  
  writeln;
  writeln('Testing another string "aaabbb":');
  InputString := 'aaabbb';
  writeln('Input string: ', InputString);
  writeln;
  
  // Perform CYK parsing for second test
  CYKParse(InputString);
  PrintCYKTable;
  
  if IsAccepted then
    writeln('Result: String "', InputString, '" is ACCEPTED by the grammar')
  else
    writeln('Result: String "', InputString, '" is REJECTED by the grammar');
  
  writeln;
  writeln('Press Enter to exit...');
  readln;
end.
```

## Key Features of this CYK Implementation:

1. **Grammar Representation**: Uses arrays to store non-terminals, terminals, and production rules
2. **CYK Table**: 3D array structure to store parsing results
3. **Algorithm Steps**:
   - Initialize diagonal elements (length 1 substrings)
   - Fill table for substrings of increasing length
   - Apply all applicable grammar rules
4. **Result Checking**: Determines if the input string is accepted by the grammar
5. **Visualization**: Prints the complete CYK parsing table
