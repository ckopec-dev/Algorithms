# Cocke-Younger-Kasami (CYK) Parsing Algorithm in C++

The CYK algorithm is a dynamic programming algorithm for parsing context-free grammars in Chomsky Normal Form (CNF). It determines whether a given string can be generated by a grammar.

## Implementation

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <sstream>

class CYKParser {
private:
    // Grammar rules in Chomsky Normal Form
    std::unordered_map<std::string, std::vector<std::string>> grammar;
    
    // Store all non-terminals
    std::unordered_set<std::string> nonTerminals;
    
    // Store all terminals
    std::unordered_set<std::string> terminals;
    
public:
    // Add a production rule A -> BC or A -> a
    void addRule(const std::string& left, const std::string& right) {
        grammar[left].push_back(right);
        
        // Track non-terminals and terminals
        if (right.length() == 1 && !isupper(right[0])) {
            terminals.insert(right);
        } else {
            nonTerminals.insert(left);
            if (right.length() == 2) {
                nonTerminals.insert(right.substr(0, 1));
                nonTerminals.insert(right.substr(1, 1));
            }
        }
    }
    
    // Check if a character is uppercase (non-terminal)
    bool isNonTerminal(const std::string& symbol) {
        return nonTerminals.find(symbol) != nonTerminals.end();
    }
    
    // Parse the input string using CYK algorithm
    bool parse(const std::string& input) {
        int n = input.length();
        
        // Create the CYK table
        std::vector<std::vector<std::unordered_set<std::string>>> table(n, 
            std::vector<std::unordered_set<std::string>>(n));
        
        // Fill the first diagonal (base case)
        for (int i = 0; i < n; i++) {
            std::string symbol(1, input[i]);
            for (const auto& rule : grammar) {
                const std::string& left = rule.first;
                const std::vector<std::string>& right = rule.second;
                
                for (const std::string& production : right) {
                    if (production.length() == 1 && production == symbol) {
                        table[i][0].insert(left);
                    }
                }
            }
        }
        
        // Fill the rest of the table
        for (int length = 2; length <= n; length++) {
            for (int i = 0; i <= n - length; i++) {
                int j = i + length - 1;
                
                for (int k = i; k < j; k++) {
                    // For each possible split at position k
                    for (const auto& rule : grammar) {
                        const std::string& left = rule.first;
                        const std::vector<std::string>& right = rule.second;
                        
                        for (const std::string& production : right) {
                            if (production.length() == 2) {
                                std::string first(1, production[0]);
                                std::string second(1, production[1]);
                                
                                // Check if both parts exist in sub-table
                                if (table[i][k-i].find(first) != table[i][k-i].end() &&
                                    table[k+1][j-(k+1)].find(second) != table[k+1][j-(k+1)].end()) {
                                    table[i][j-i].insert(left);
                                }
                            }
                        }
                    }
                }
            }
        }
        
        // Check if start symbol (assumed to be S) is in the top cell
        return table[0][n-1].find("S") != table[0][n-1].end();
    }
    
    // Print the parsing table for debugging
    void printTable(const std::string& input) {
        int n = input.length();
        std::vector<std::vector<std::unordered_set<std::string>>> table(n, 
            std::vector<std::unordered_set<std::string>>(n));
        
        // Fill the first diagonal
        for (int i = 0; i < n; i++) {
            std::string symbol(1, input[i]);
            for (const auto& rule : grammar) {
                const std::string& left = rule.first;
                const std::vector<std::string>& right = rule.second;
                
                for (const std::string& production : right) {
                    if (production.length() == 1 && production == symbol) {
                        table[i][0].insert(left);
                    }
                }
            }
        }
        
        // Fill the rest of the table
        for (int length = 2; length <= n; length++) {
            for (int i = 0; i <= n - length; i++) {
                int j = i + length - 1;
                
                for (int k = i; k < j; k++) {
                    for (const auto& rule : grammar) {
                        const std::string& left = rule.first;
                        const std::vector<std::string>& right = rule.second;
                        
                        for (const std::string& production : right) {
                            if (production.length() == 2) {
                                std::string first(1, production[0]);
                                std::string second(1, production[1]);
                                
                                if (table[i][k-i].find(first) != table[i][k-i].end() &&
                                    table[k+1][j-(k+1)].find(second) != table[k+1][j-(k+1)].end()) {
                                    table[i][j-i].insert(left);
                                }
                            }
                        }
                    }
                }
            }
        }
        
        // Print the table
        std::cout << "CYK Parsing Table for input \"" << input << "\":" << std::endl;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (j >= i) {
                    std::cout << "{";
                    int count = 0;
                    for (const auto& symbol : table[i][j-i]) {
                        if (count > 0) std::cout << ",";
                        std::cout << symbol;
                        count++;
                    }
                    std::cout << "} ";
                } else {
                    std::cout << "      ";
                }
            }
            std::cout << std::endl;
        }
        std::cout << std::endl;
    }
};

int main() {
    CYKParser parser;
    
    // Define a simple grammar in Chomsky Normal Form
    // S -> AB | BC
    // A -> BA | a
    // B -> CC | b
    // C -> AB | a
    
    parser.addRule("S", "AB");
    parser.addRule("S", "BC");
    parser.addRule("A", "BA");
    parser.addRule("A", "a");
    parser.addRule("B", "CC");
    parser.addRule("B", "b");
    parser.addRule("C", "AB");
    parser.addRule("C", "a");
    
    // Test cases
    std::vector<std::string> testCases = {"ab", "ba", "aa", "bb", "aba", "bab"};
    
    std::cout << "CYK Parsing Algorithm Demo" << std::endl;
    std::cout << "===========================" << std::endl;
    
    for (const std::string& test : testCases) {
        bool result = parser.parse(test);
        std::cout << "Input: \"" << test << "\" -> " 
                  << (result ? "ACCEPTED" : "REJECTED") << std::endl;
    }
    
    std::cout << "\nDetailed parsing for \"aba\":" << std::endl;
    parser.printTable("aba");
    
    return 0;
}
```

## Algorithm Explanation

1. **Initialization**: Fill the diagonal of the parsing table with terminals that match each character in the input string.

2. **Dynamic Programming**: For each substring length from 2 to n:
   - Try all possible splits of the substring
   - For each split, check if any grammar rule can be applied
   - Add the resulting non-terminal to the table cell

3. **Result**: The input string is accepted if the start symbol (S) appears in the top-right cell of the table.

## Time and Space Complexity

- **Time Complexity**: O(n³ × |G|) where n is the input length and |G| is the number of grammar rules
- **Space Complexity**: O(n² × |N|) where |N| is the number of non-terminals

The CYK algorithm is particularly useful for parsing context-free grammars and is the foundation for many practical parsing applications.
