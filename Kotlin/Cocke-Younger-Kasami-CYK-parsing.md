# Cocke-Younger-Kasami (CYK) Parsing Algorithm in Kotlin

The CYK algorithm is a dynamic programming algorithm for parsing context-free grammars in Chomsky Normal Form (CNF). It determines whether a string can be generated by a given grammar.

## Implementation

```kotlin
import java.util.*

data class Production(val left: String, val right: List<String>)

class CYKParser {
    private val grammar: Map<String, List<List<String>>> = mapOf(
        "S" to listOf(listOf("A", "B"), listOf("B", "C")),
        "A" to listOf(listOf("a")),
        "B" to listOf(listOf("b")),
        "C" to listOf(listOf("c"))
    )
    
    private val nonTerminals = grammar.keys.toSet()
    
    fun parse(input: String): Boolean {
        val n = input.length
        if (n == 0) return false
        
        // Create the parsing table
        val table = Array(n) { Array(n) { mutableSetOf<String>() } }
        
        // Fill the first diagonal (base case)
        for (i in 0 until n) {
            val symbol = input[i].toString()
            for ((nonTerminal, productions) in grammar) {
                if (productions.any { it.size == 1 && it[0] == symbol }) {
                    table[i][i].add(nonTerminal)
                }
            }
        }
        
        // Fill the rest of the table
        for (length in 2..n) {
            for (i in 0..n - length) {
                val j = i + length - 1
                
                for (k in i until j) {
                    // For each possible split point k
                    for ((nonTerminal, productions) in grammar) {
                        for (production in productions) {
                            if (production.size == 2) {
                                val left = production[0]
                                val right = production[1]
                                
                                if (table[i][k].contains(left) && 
                                    table[k + 1][j].contains(right)) {
                                    table[i][j].add(nonTerminal)
                                }
                            }
                        }
                    }
                }
            }
        }
        
        // Check if the start symbol (S) can generate the entire string
        return table[0][n - 1].contains("S")
    }
    
    fun parseWithTrace(input: String): Boolean {
        val n = input.length
        if (n == 0) return false
        
        val table = Array(n) { Array(n) { mutableSetOf<String>() } }
        
        println("Parsing input: $input")
        println("Grammar:")
        grammar.forEach { (nonTerminal, productions) ->
            println("  $nonTerminal → ${productions.joinToString(" | ")}")
        }
        println()
        
        // Fill the first diagonal
        println("Step 1: Initialize diagonal")
        for (i in 0 until n) {
            val symbol = input[i].toString()
            for ((nonTerminal, productions) in grammar) {
                if (productions.any { it.size == 1 && it[0] == symbol }) {
                    table[i][i].add(nonTerminal)
                    println("  Position $i: '$symbol' can be derived from $nonTerminal")
                }
            }
        }
        println()
        
        // Fill the rest of the table
        println("Step 2: Fill the table using dynamic programming")
        for (length in 2..n) {
            println("  Length $length:")
            for (i in 0..n - length) {
                val j = i + length - 1
                println("    From position $i to $j:")
                
                for (k in i until j) {
                    for ((nonTerminal, productions) in grammar) {
                        for (production in productions) {
                            if (production.size == 2) {
                                val left = production[0]
                                val right = production[1]
                                
                                if (table[i][k].contains(left) && 
                                    table[k + 1][j].contains(right)) {
                                    table[i][j].add(nonTerminal)
                                    println("      $nonTerminal → $left $right (positions $i-$k and $k+1-$j)")
                                }
                            }
                        }
                    }
                }
            }
        }
        println()
        
        val result = table[0][n - 1].contains("S")
        println("Result: ${if (result) "String can be generated by grammar" else "String cannot be generated by grammar"}")
        println("Final table:")
        for (i in 0 until n) {
            for (j in 0 until n) {
                if (table[i][j].isNotEmpty()) {
                    print("  [$i,$j] = ${table[i][j]}  ")
                }
            }
            println()
        }
        
        return result
    }
}

fun main() {
    val parser = CYKParser()
    
    // Test cases
    val testCases = listOf("abc", "ab", "ac", "a", "b", "c", "abcc", "")
    
    println("=== CYK Parser Examples ===")
    println()
    
    for (testCase in testCases) {
        println("Testing string: \"$testCase\"")
        val result = parser.parse(testCase)
        println("Result: $result")
        println()
    }
    
    println("=== Detailed Trace Example ===")
    println()
    parser.parseWithTrace("abc")
}
```

## How It Works

1. **Initialization**: Fill the diagonal of the parsing table with non-terminals that can generate each individual symbol
2. **Dynamic Programming**: For each substring length from 2 to n:
   - Try all possible splits of the substring
   - For each split, check if any production can be applied
   - Add the resulting non-terminal to the table cell
3. **Final Check**: The string is in the language if the start symbol S appears in the top-right cell

## Time and Space Complexity

- **Time Complexity**: O(n³) where n is the length of the input string
- **Space Complexity**: O(n²) for the parsing table

## Key Features

- Handles context-free grammars in Chomsky Normal Form
- Uses dynamic programming for efficient parsing
- Provides detailed trace output for educational purposes
- Works with any grammar that can be converted to CNF

This implementation demonstrates the core principles of the CYK algorithm while providing clear output for understanding how the parsing process works.
