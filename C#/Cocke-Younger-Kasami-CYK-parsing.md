# Cocke-Younger-Kasami (CYK) Parsing Algorithm in C#

The CYK algorithm is a dynamic programming algorithm for parsing context-free grammars in Chomsky Normal Form (CNF). It determines whether a string can be generated by a given grammar.

## Implementation

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public class CYKParser
{
    private Dictionary<string, List<string>> grammar;
    private HashSet<string> nonTerminals;
    
    public CYKParser(Dictionary<string, List<string>> grammar)
    {
        this.grammar = grammar;
        this.nonTerminals = new HashSet<string>(grammar.Keys);
    }
    
    /// <summary>
    /// Parse a string using CYK algorithm
    /// </summary>
    /// <param name="input">Input string to parse</param>
    /// <returns>True if string can be generated by grammar, false otherwise</returns>
    public bool Parse(string input)
    {
        if (string.IsNullOrEmpty(input))
            return false;
            
        int n = input.Length;
        
        // Create the parsing table
        Dictionary<(int, int), HashSet<string>> table = new Dictionary<(int, int), HashSet<string>>();
        
        // Initialize the first diagonal (base case)
        for (int i = 0; i < n; i++)
        {
            char symbol = input[i];
            table[(i, 1)] = new HashSet<string>();
            
            // Find all non-terminals that produce this terminal
            foreach (var rule in grammar)
            {
                if (rule.Value.Contains(symbol.ToString()))
                {
                    table[(i, 1)].Add(rule.Key);
                }
            }
        }
        
        // Fill the table using dynamic programming
        for (int length = 2; length <= n; length++)
        {
            for (int i = 0; i <= n - length; i++)
            {
                table[(i, length)] = new HashSet<string>();
                
                // Try all possible partitions
                for (int k = 1; k < length; k++)
                {
                    int j = i + k;
                    
                    // For each rule A -> BC in the grammar
                    foreach (var rule in grammar)
                    {
                        string leftSide = rule.Key;
                        var rightSides = rule.Value;
                        
                        foreach (string rightSide in rightSides)
                        {
                            // Check if right side is of the form "BC"
                            if (rightSide.Length == 2)
                            {
                                char first = rightSide[0];
                                char second = rightSide[1];
                                
                                // Check if we can derive from left and right parts
                                if (table.ContainsKey((i, k)) && 
                                    table.ContainsKey((j, length - k)))
                                {
                                    if (table[(i, k)].Contains(first.ToString()) && 
                                        table[(j, length - k)].Contains(second.ToString()))
                                    {
                                        table[(i, length)].Add(leftSide);
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
        
        // Check if start symbol can derive the entire string
        return table.ContainsKey((0, n)) && table[(0, n)].Contains("S");
    }
    
    /// <summary>
    /// Get all possible derivations for a given substring
    /// </summary>
    public HashSet<string> GetDerivations(string input)
    {
        if (string.IsNullOrEmpty(input))
            return new HashSet<string>();
            
        int n = input.Length;
        Dictionary<(int, int), HashSet<string>> table = new Dictionary<(int, int), HashSet<string>>();
        
        // Initialize the first diagonal
        for (int i = 0; i < n; i++)
        {
            char symbol = input[i];
            table[(i, 1)] = new HashSet<string>();
            
            foreach (var rule in grammar)
            {
                if (rule.Value.Contains(symbol.ToString()))
                {
                    table[(i, 1)].Add(rule.Key);
                }
            }
        }
        
        // Fill the table
        for (int length = 2; length <= n; length++)
        {
            for (int i = 0; i <= n - length; i++)
            {
                table[(i, length)] = new HashSet<string>();
                
                for (int k = 1; k < length; k++)
                {
                    int j = i + k;
                    
                    foreach (var rule in grammar)
                    {
                        string leftSide = rule.Key;
                        var rightSides = rule.Value;
                        
                        foreach (string rightSide in rightSides)
                        {
                            if (rightSide.Length == 2)
                            {
                                char first = rightSide[0];
                                char second = rightSide[1];
                                
                                if (table.ContainsKey((i, k)) && 
                                    table.ContainsKey((j, length - k)))
                                {
                                    if (table[(i, k)].Contains(first.ToString()) && 
                                        table[(j, length - k)].Contains(second.ToString()))
                                    {
                                        table[(i, length)].Add(leftSide);
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
        
        return table.ContainsKey((0, n)) ? table[(0, n)] : new HashSet<string>();
    }
}

// Example usage
public class Program
{
    public static void Main()
    {
        // Define a grammar in Chomsky Normal Form
        // S -> AB | BC
        // A -> BA | a
        // B -> CC | b
        // C -> AB | a
        
        var grammar = new Dictionary<string, List<string>>
        {
            { "S", new List<string> { "AB", "BC" } },
            { "A", new List<string> { "BA", "a" } },
            { "B", new List<string> { "CC", "b" } },
            { "C", new List<string> { "AB", "a" } }
        };
        
        CYKParser parser = new CYKParser(grammar);
        
        // Test cases
        string[] testStrings = { "ab", "aab", "baa", "abb", "aaa" };
        
        Console.WriteLine("CYK Parser Results:");
        Console.WriteLine("===================");
        
        foreach (string test in testStrings)
        {
            bool result = parser.Parse(test);
            var derivations = parser.GetDerivations(test);
            
            Console.WriteLine($"String: \"{test}\" -> {(result ? "Accepted" : "Rejected")}");
            if (derivations.Count > 0)
            {
                Console.WriteLine($"  Derivations: {string.Join(", ", derivations)}");
            }
            Console.WriteLine();
        }
    }
}
```

## Example Grammar Analysis

For the grammar defined in the example:
```
S → AB | BC
A → BA | a  
B → CC | b
C → AB | a
```

This grammar generates strings like:
- "ab" (S → AB → aB → ab)
- "aab" (S → AB → aB → aaB → aab)
- "baa" (S → BC → bA → baA → baa)

## Algorithm Complexity

- **Time Complexity**: O(n³ × |G|) where n is the string length and |G| is the number of grammar rules
- **Space Complexity**: O(n² × |N|) where |N| is the number of non-terminals

## Key Features

1. **Chomsky Normal Form Support**: Works with grammars in CNF format
2. **Dynamic Programming**: Uses table-filling approach for efficient parsing
3. **Membership Testing**: Determines if a string belongs to the language
4. **Derivation Tracking**: Can identify possible derivations for valid strings

The CYK algorithm is particularly useful for:
- Natural language processing
- Compiler design
- Grammar validation
- Parsing context-free languages efficiently
