  "Fast inverse square root implementation using bit manipulation"
  (declare (single-float x))
  (let ((i (float->uint (single-float x))))
    (let ((i (logand (logior (ash i -1) #x3FFFFFFF) #x7FFFFFFF)))
      (let ((y (uint->float i)))
        (let ((y (single-float y)))
          (let ((y (* y (- 1.5 (single-float (* 0.5 x y y)))))
            (let ((y (* y (- 1.5 (single-float (* 0.5 x y y)))))
              (let ((y (* y (- 1.5 (single-float (* 0.5 x y y)))))
                (single-float y))))))))))

(defun float->uint (f)
  "Convert single-float to unsigned integer using bit manipulation"
  (declare (single-float f))
  (the unsigned-byte
    (ldb (byte 32 0) (sxhash f))))

(defun uint->float (i)
  "Convert unsigned integer to single-float using bit manipulation"
  (declare (unsigned-byte i))
  (the single-float
    (let ((f (make-array 1 :element-type 'single-float)))
      (setf (aref f 0) (coerce i 'single-float))
      (aref f 0))))
